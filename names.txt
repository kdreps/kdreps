    const CAT_NAMES = {
      'buty': 'Buty', 'koszulki-bluzy': 'Koszulki & Bluzy', 'koszulki-pilkarskie': 'Koszulki Piłkarskie',

    const CAT_ICONS = {
      'buty': '👟', 'koszulki-bluzy': '👕', 'koszulki-pilkarskie': '⚽', 'spodenki': '🩳',
      'spodnie': '👖', 'sety': '👔', 'skarpetki': '🧦', 'majtki': '🩲', 'paski': '🥋',
      'portfele': '👛', 'plecaki': '🎒', 'czapki': '🧢', 'kurtki': '🧥', 'perfumy': '💧'
    };

    const norm = s => (s || '').normalize('NFD').replace(/[\u0300-\u036f]/g, '').toLowerCase();

    function scoreProduct(item, nq, words) {
      const n = norm(item.name);
      const cat = norm(CAT_NAMES[item.category] || item.category);
      let score = 0;
      // Exact match gets highest score
      if (n === nq) score += 100;
      // Starts with query
      if (n.startsWith(nq)) score += 50;
      // Contains full query as substring
      if (n.includes(nq)) score += 30;
      // Category match
      if (cat.includes(nq)) score += 20;
      // Individual word matches (fuzzy)
      words.forEach(w => {
        if (w.length < 2) return;
        if (n.includes(w)) score += 15;
        if (cat.includes(w)) score += 5;
      });
      return score;
    }

    function handleSearch(val) {
      const q = val.trim();
      const box = document.getElementById('searchResults');
      if (q.length < 2) { box.classList.remove('visible'); box.innerHTML = ''; return; }

      const nq = norm(q);
      const words = nq.split(/\s+/).filter(w => w.length >= 2);

      if (typeof KAKOBUY_DB === 'undefined') return;

      // Score and filter products
      const results = KAKOBUY_DB
        .map((item, idx) => ({ item, idx, score: scoreProduct(item, nq, words) }))
        .filter(r => r.score > 0)
        .sort((a, b) => b.score - a.score)
        .slice(0, 12);

      if (!results.length) {
        box.innerHTML = `<div class="search-result-item"><div class="sri-text"><div class="sri-name" style="color:var(--text3)">Brak wyników dla &ldquo;${q}&rdquo;</div><div class="sri-sub">Spróbuj innej nazwy lub marki</div></div></div>`;
      } else {
        const header = `<div class="search-section-header">📦 PRODUKTY</div>`;
        const items = results.map(({ item, idx }) => {
          const thumb = item.image
            ? `<img src="${item.image}" alt="${item.name}">`
            : `<div class="sri-thumb-emoji">${CAT_ICONS[item.category] || '🛍️'}</div>`;
          const price = item.pricePLN ? `<span class="sri-price">${item.pricePLN} zł</span>` : '';
          const catLabel = CAT_NAMES[item.category] || item.category;
          return `
          <div class="search-result-item" onclick="openProduct(${idx}, '${item.category}'); clearSearch();">
            <div class="sri-thumb">${thumb}</div>
            <div class="sri-text">
              <div class="sri-name">${item.name}</div>
              <div class="sri-sub">${price}<span class="sri-cat">${catLabel}</span></div>
            </div>
            <div class="sri-arrow">→</div>
          </div>`;
        }).join('');
        const footer = results.length === 12
          ? `<div class="search-result-count">⚡ Wyświetlono 12 / więcej – Wpisz dokładniejszą frazę</div>` : '';
        box.innerHTML = header + items + footer;
      }
      box.classList.add('visible');
    }

    function doSearch() {
      const q = document.getElementById('searchInput').value.trim();
      if (!q || typeof KAKOBUY_DB === 'undefined') return;
      const nq = norm(q);
      const words = nq.split(/\s+/).filter(w => w.length >= 2);
      const best = KAKOBUY_DB
        .map((item, idx) => ({ idx, score: scoreProduct(item, nq, words) }))
        .filter(r => r.score > 0)
        .sort((a, b) => b.score - a.score)[0];
      if (best) { openProduct(best.idx, KAKOBUY_DB[best.idx].category); clearSearch(); }
    }

      const from = document.getElementById('from').value;
      const to = document.getElementById('to').value;
      const weight = parseFloat(document.getElementById('weight').value) || 1;
      const type = document.getElementById('type').value;
      const insurance = parseInt(document.getElementById('insurance').value) || 0;
      const cod = parseInt(document.getElementById('cod').value) || 0;
      const promoCode = document.getElementById('promoInput').value.trim().toUpperCase();

      const dest = to === 'pl' ? from : to;
      const base = RATES.base[dest] * RATES.type[type];
      const weightCost = weight * 2.5 * RATES.type[type];
      const subtotal = base + weightCost + insurance + cod;

      let discount = 1, promoLabel = '';
      if (promoCode && PROMOS[promoCode]) { discount = PROMOS[promoCode]; promoLabel = ' (kod: ' + promoCode + ')'; }
      else if (promoCode) { promoLabel = ' (nieprawidłowy kod)'; }

      const total = (subtotal * discount).toFixed(2);
      const savings = promoCode && PROMOS[promoCode] ? (subtotal * (1 - discount)).toFixed(2) : 0;

      document.getElementById('resultPrice').textContent = total + ' zł';
      const typeNames = { std: 'Standardowa', exp: 'Ekspresowa', eco: 'Ekonomiczna', pal: 'Paleta' };
      let bd = `<div class="result-line">Baza: <strong>${base.toFixed(2)} zł</strong></div>
      <div class="result-line">Waga ${weight}kg: <strong>${weightCost.toFixed(2)} zł</strong></div>
      <div class="result-line">Typ: <strong>${typeNames[type]}</strong></div>`;
      if (insurance) bd += `<div class="result-line">Ubezpieczenie: <strong>+${insurance} zł</strong></div>`;
      if (cod) bd += `<div class="result-line">Pobranie: <strong>+${cod} zł</strong></div>`;
      if (savings > 0) bd += `<div class="result-line" style="color:var(--accent)">Oszczędzasz: <strong>-${savings} zł${promoLabel}</strong></div>`;
      else if (promoLabel) bd += `<div class="result-line" style="color:var(--red)">${promoLabel}</div>`;
      document.getElementById('resultBreakdown').innerHTML = bd;
    }
  </script>
